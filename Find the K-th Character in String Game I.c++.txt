class Solution {
public:
    char kthCharacter(int k) {
        string original="a";
        string addi="b";
        while(original.size()<=k){
            original+=addi;
            string sub="";
            int i=0;
            while(i<addi.size()){
                if(addi[i]=='z') sub+='a';
                else 
                sub+=(addi[i]+1);
                i++;
            }
            addi+=sub;
        }
        return original[k-1];
    }
};
