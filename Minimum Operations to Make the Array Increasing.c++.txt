class Solution {
public:
    int minOperations(vector<int>& nums) {
        if(nums.size()==1) return 0;
        int cnt=0;
        for(int i=0;i<nums.size()-1;i++){
            if(nums[i]>=nums[i+1]){
                int inc=nums[i]-nums[i+1]+1;
                cnt+=inc;
                nums[i+1]=nums[i+1]+inc;
            }

        }
        int s=nums.size()-1;
        if(nums[s]<=nums[s-1]) cnt+=(nums[s-1]-nums[s]+1);
        return cnt;
    }
};
