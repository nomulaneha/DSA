class Solution {
public:
    bool hasSameDigits(string s) {
        while(s.size()>2){
            string temp="";
            for(int i=0;i<s.size()-1;i++){
                int temp1=s[i]-'0';
                int temp2=s[i+1]-'0';
                int rem = (temp1+temp2)%10;
                temp+=to_string(rem);
            } 
            s=temp;
        }
        if(s[0]==s[1]) return true;
        else return false;
    }
};
