class Solution {
public:
    string restoreString(string s, vector<int>& indices) {
        map<int, char>mpp;
        int j=0;
        for(int i=0;i<indices.size();i++){
            mpp[indices[i]]=s[j];
            j++;
        }
        string res="";
        for(auto &it:mpp){
            res+=it.second;
        }
        return res;
    }
};
