class Solution {
public:
    vector<int> decompressRLElist(vector<int>& nums) {
        vector<int>res;
        for(int i=0;i<nums.size();i++){
            if(2*i<nums.size() && (2*i)+1<nums.size()){
                int freq = nums[2*i];
                int val = nums[2*i+1];
                res.insert(res.end(), freq, val);
            }
            else{
                break;
            }
        }
        return res;
    }
};
