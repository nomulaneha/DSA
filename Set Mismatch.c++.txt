class Solution {
public:
    vector<int> findErrorNums(vector<int>& nums) {
        vector<int>res(2);
        vector<int>ar(nums.size()+1,0);
        ar[0]=-1;
        for(int i=0;i<nums.size();i++){
           ar[nums[i]]+=1;
        }
        for(int i=1;i<=nums.size();i++){
            if(ar[i]==2) res[0]=i;
            if(ar[i]==0) res[1]=i;
        }
        return res;
    }
};
