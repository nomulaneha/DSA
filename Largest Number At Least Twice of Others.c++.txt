class Solution {
public:
    int dominantIndex(vector<int>& nums) {
        int maxi = *max_element(nums.begin(), nums.end());
        int cnt=0;
        int ind=-1;
        for(int i=0;i<nums.size();i++){
            if(nums[i]==maxi) ind=i;
            if(maxi>=2*nums[i]) cnt++;
        }
        if(cnt==nums.size()-1) return ind;
        else return -1;
    }
};
