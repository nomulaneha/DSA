class Solution {
public:
    int prefixCount(vector<string>& words, string pref) {
        int len=pref.size();
        int cnt=0;
        for(int i=0;i<words.size();i++){
            string s="";
            for(int j=0;j<len;j++){
                s+=words[i][j];
            }
            if(s==pref) cnt++;
        }
        return cnt;
    }
};
