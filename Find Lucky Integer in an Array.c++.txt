class Solution {
public:
    int findLucky(vector<int>& arr) {
        map<int,int>mpp;
        for(int i=0;i<arr.size();i++){
            mpp[arr[i]]+=1;
        }
        int res=-1;
        for(auto &it:mpp){
            if(it.first==it.second) res=it.first;
        }
        return res;
    }
};
